#!/bin/sh
doas apk add pipewire wireplumber
doas apk add pipewire-pulse
doas apk add pipewire-jack
doas apk add pipewire-alsa
doas apk add bluez bluez-openrc pipewire-spa-bluez

rc-service --user wireplumber start
rc-update  --user add pipewire    default
rc-update  --user add wireplumber default

doas addgroup "$USER" audio
doas addgroup "$USER" video
