#!/bin/sh
./linux.txt
if ! [ -e /etc/gentoo-release ]; then
	>2 echo "Not a Gentoo system"
	exit 1
fi
doas mkdir -p /etc/portage/{package.use,sets,package.mask,package.accept_keywords}

doas rm -v /etc/portage/package.*/*
doas rm -v /etc/portage/sets/*
doas cp -v etc-Gentoo/portage/package.use/* /etc/portage/package.use/
doas cp -v etc-Gentoo/portage/package.accept_keywords/* \
	/etc/portage/package.accept_keywords/
doas cp -v etc-Gentoo/portage/package.mask/* \
	/etc/portage/package.mask/

doas cp -v etc-Gentoo/portage/sets/* /etc/portage/sets/
doas cp -v etc-Gentoo/portage/*.conf /etc/portage/
